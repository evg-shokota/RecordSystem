# plugins/my_module/plugin.py
from core.plugin_base import BasePlugin

class Plugin(BasePlugin):
    name        = "Мій модуль"
    slug        = "my_module"
    version     = "1.0.0"
    description = "Короткий опис того що робить модуль"
    author      = "Ваше ім'я"
    icon        = "bi-puzzle"

    def register(self, app, api, hooks=None):
        self.api = api
        from .routes import bp
        bp.api = api
        app.register_blueprint(bp)

{# Filter hooks #}
Filter hooks — filter_*

Змінити дані до відображення. Отримує значення, повертає змінене.

МетодЩо фільтруєтьсяТип значення
filter_personnel__display_nameВідображуване ім'я о/сstr
filter_invoice__can_deleteЧи можна видалити накладнуbool
filter_sidebar__menuСписок пунктів бічного менюlist[dict]
filter_dashboard__statsСтатистика дашбордуdict
 
def filter_personnel__display_name(self, value, person=None, **kwargs):
    """Змінити відображуване ім'я о/с."""
    # value — поточний рядок, напр. "Іваненко І.І."
    if person and person.get('rank') == 'генерал':
        return f"★ {value}"
    return value

def filter_invoice__can_delete(self, value, invoice=None, **kwargs):
    """Заборонити видалення затверджених накладних."""
    if invoice and invoice.get('status') == 'approved':
        return False
    return value

# plugins/my_module/routes.py
from flask import Blueprint, render_template, request, jsonify
from core.auth import login_required

bp = Blueprint("my_module", __name__, url_prefix="/my-module",
               template_folder="templates")
bp.api = None  # встановлюється в plugin.register()


@bp.route("/")
@login_required
def index():
    people = bp.api.personnel.get_list(is_active=True)
    return render_template("my_module/index.html", people=people)


@bp.route("/api/data")
@login_required
def api_data():
    rows = bp.api.db.execute("SELECT * FROM my_table LIMIT 100")
    return jsonify({"ok": True, "data": rows})


@bp.route("/action", methods=["POST"])
@login_required
def action():
    data = request.get_json()
    new_id = bp.api.db.write(
        "INSERT INTO my_table (value) VALUES (?)", (data['value'],)
    )
    bp.api.audit.log('create', 'my_table', new_id, new_data=data)
    return jsonify({"ok": True, "id": new_id})
